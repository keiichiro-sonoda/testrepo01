\section{}
$\mathbb{C}^2$の標準基底
\[
  e_1 = \begin{bmatrix}
    1 \\ 0
  \end{bmatrix}, 
  e_2 = \begin{bmatrix}
    0 \\ 1
  \end{bmatrix}
\]
で$\{{\rm H}e_1, {\rm H}e_2\}$がCONSになることを示す.
\begin{eqnarray*}
  {\rm H}e_1 = \frac{1}{\sqrt{2}} \begin{bmatrix}
    1 & 1 \\ 1 & -1
  \end{bmatrix} \begin{bmatrix}
    1 \\ 0
  \end{bmatrix} = \frac{1}{\sqrt{2}} \begin{bmatrix}
    1 \\ 1
  \end{bmatrix} \\
  {\rm H}e_2 = \frac{1}{\sqrt{2}} \begin{bmatrix}
    1 & 1 \\ 1 & -1
  \end{bmatrix} \begin{bmatrix}
    0 \\ 1
  \end{bmatrix} = \frac{1}{\sqrt{2}} \begin{bmatrix}
    1 \\ -1
  \end{bmatrix}
\end{eqnarray*}
